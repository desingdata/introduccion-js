Proyecto Práctico · ES6+

Conjuntos en JavaScript

// Estructura Set — colección de valores únicos sin duplicados

01
Creación de un Set
// Un Set almacena valores únicos de cualquier tipo. Los duplicados se ignoran automáticamente.
código
// new Set() — crea un conjunto vacío o desde iterable
const frutas = new Set([
  'manzana', 'pera', 'mango',
  'manzana', 'pera'  // ← duplicados
]);

// .size — propiedad que retorna el número de elementos
console.log(frutas.size);

// El Set elimina automáticamente los duplicados
console.log(frutas);
salida
3
Set(3) { 'manzana', 'pera', 'mango' }
02
Métodos: add, has, delete, clear
// Operaciones básicas para gestionar los elementos del conjunto.
código
const nums = new Set([1, 2, 3]);

// .add(valor) — agrega un elemento; retorna el mismo Set
nums.add(4).add(5);

// .has(valor) — verifica si el elemento existe (boolean)
console.log(nums.has(3));  // true
console.log(nums.has(9));  // false

// .delete(valor) — elimina un elemento específico
nums.delete(2);
console.log(nums);

// .clear() — vacía todo el conjunto
const temp = new Set([10, 20]);
temp.clear();
console.log(temp.size);
salida
true
false
Set(4) { 1, 3, 4, 5 }
0
03
Iteración: for...of, forEach, spread
// Un Set es iterable; se puede recorrer o convertir a array fácilmente.
código
const colores = new Set(['rojo','verde','azul']);

// for...of — recorre cada valor del conjunto
for (const color of colores) {
  console.log(`Color: ${color}`);
}

// .forEach(callback) — igual que en arrays
colores.forEach(c => console.log(c.toUpperCase()));

// Spread operator [...set] — convierte Set a Array
const arr = [...colores];
console.log(arr);

// Array.from(set) — otra forma de convertir
const arr2 = Array.from(colores);
console.log(arr2[0]);
salida
Color: rojo
Color: verde
Color: azul
ROJO VERDE AZUL
['rojo', 'verde', 'azul']
'rojo'
04
Caso de uso: Eliminar duplicados de un Array
// El Set es la forma más elegante y eficiente de limpiar arrays con repetidos.
código
const datos = [5,3,8,3,1,5,9,8,2];

// Conversión: Array → Set → Array
// El Set descarta los repetidos al construirse
const sinDuplicados = [...new Set(datos)];
console.log(sinDuplicados);

// Función genérica reutilizable
const unique = arr => [...new Set(arr)];

const tags = ['js','css','js','html','css'];
console.log(unique(tags));
salida
[5, 3, 8, 1, 9, 2]
['js', 'css', 'html']
05
Operaciones matemáticas: Unión, Intersección, Diferencia
// Algebra de conjuntos: combinar, intersectar o restar dos Sets como en matemáticas.
código
const A = new Set([1,2,3,4]);
const B = new Set([3,4,5,6]);

// UNIÓN — todos los elementos de A y B (sin repetir)
const union = new Set([...A, ...B]);
console.log('Unión:', union);

// INTERSECCIÓN — solo elementos en ambos conjuntos
const interseccion = new Set(
  [...A].filter(x => B.has(x))
);
console.log('Intersección:', interseccion);

// DIFERENCIA — elementos de A que NO están en B
const diferencia = new Set(
  [...A].filter(x => !B.has(x))
);
console.log('Diferencia A-B:', diferencia);
salida
Unión: Set(6) { 1, 2, 3, 4, 5, 6 }
Intersección: Set(2) { 3, 4 }
Diferencia A-B: Set(2) { 1, 2 }
06
WeakSet: referencias débiles a objetos
// WeakSet solo acepta objetos y no impide que sean eliminados por el Garbage Collector.
código
// WeakSet — conjunto de objetos con referencias débiles
// No tiene .size ni es iterable (por diseño)
const visitados = new WeakSet();

let nodo1 = { id: 1, nombre: 'Home' };
let nodo2 = { id: 2, nombre: 'About' };

// .add() solo acepta objetos
visitados.add(nodo1);
visitados.add(nodo2);

// .has() — verifica si el objeto fue agregado
console.log(visitados.has(nodo1)); // true

// Si nodo1 = null, el GC puede liberarlo
// El WeakSet no bloquea la recolección de basura
nodo1 = null;
console.log('nodo2 visitado:', visitados.has(nodo2));
salida
true
nodo2 visitado: true
// nodo1 queda disponible para GC

new Set([iter])
Crea un conjunto, opcionalmente desde un iterable. Elimina duplicados.
.add(valor)
Agrega un elemento. Si ya existe, no hace nada. Retorna el Set.
.has(valor)
Devuelve true si el elemento pertenece al conjunto.
.delete(valor)
Elimina el elemento indicado. Retorna true si existía.
.clear()
Vacía completamente el conjunto.
.size
Propiedad (no método) con la cantidad de elementos únicos.
.forEach(cb)
Itera cada valor ejecutando el callback, igual que en Array.
[...set]
Spread operator: convierte el Set en un Array estándar.
