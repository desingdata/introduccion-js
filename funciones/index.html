01 Declaración de función function declaration
01-declaracion.js 
// Declaración de función: se define con la palabra clave 'function'
// seguida del nombre. Se puede llamar ANTES de su definición (hoisting).

function saludar(nombre) {
  // 'nombre' es el parámetro: valor que recibe la función al llamarse
  return `¡Hola, ${nombre}! Bienvenido a JavaScript.`;
}

// Llamada a la función pasando un argumento
const mensaje = saludar("Ana");
console.log(mensaje); // ¡Hola, Ana! Bienvenido a JavaScript.
Declaración (Function Declaration): Forma clásica de crear funciones. Gracias al hoisting, el motor de JS la mueve al inicio del scope, por lo que se puede invocar antes de escribirla en el código.

02 Expresión de función function expression
02-expresion.js 
// Expresión de función: se asigna a una variable.
// NO tiene hoisting, solo puede usarse después de su definición.

const calcularArea = function(base, altura) {
  // La función recibe dos parámetros y devuelve su producto / 2
  return (base * altura) / 2;
};

const area = calcularArea(10, 5);
console.log(`Área del triángulo: ${area}`); // 25
Expresión de función (Function Expression): La función se trata como un valor y se guarda en una variable. No se eleva con hoisting. Es útil para pasarla como argumento o asignarla condicionalmente.

03 Arrow Function sintaxis flecha ⇒
03-arrow.js 
// Arrow function: sintaxis compacta introducida en ES6.
// Si solo hay una expresión, el 'return' es implícito.

// Forma completa con bloque de código
const cuadrado = (n) => {
  return n ** 2;
};

// Forma compacta: retorno implícito (sin llaves)
const doble = n => n * 2;

// Con arrays: las arrows son ideales con métodos de orden superior
const nums = [1, 2, 3, 4, 5];
const cuadrados = nums.map(n => n ** 2);
console.log(cuadrados); // [1, 4, 9, 16, 25]
Arrow Function: Sintaxis corta y moderna. No tiene su propio this, lo hereda del contexto donde fue creada. Ideal para callbacks y métodos funcionales como map, filter, reduce.

04 Parámetros default · rest · spread
04-parametros.js 
// Parámetro por defecto: valor usado si no se pasa argumento
const potencia = (base, exp = 2) => base ** exp;
console.log(potencia(3));    // 9 (usa exp=2 por defecto)
console.log(potencia(3, 3)); // 27

// Parámetro rest (...): agrupa el resto de argumentos en un array
const sumarTodo = (...numeros) =>
  numeros.reduce((acc, n) => acc + n, 0);

console.log(sumarTodo(1, 2, 3, 4)); // 10

// Spread: expande un array como argumentos individuales
const valores = [5, 8, 3];
console.log(Math.max(...valores)); // 8
Parámetros avanzados: Los default params evitan errores si falta un argumento. El operador rest (...param) captura argumentos variables. El operador spread (...) expande arrays al invocar funciones.

05 Scope ámbito de variables
05-scope.js 
// Scope global: disponible en todo el programa
let global = "soy global";

function exterior() {
  // Scope de función: solo accesible dentro de 'exterior'
  let local = "soy local";

  function interior() {
    // Scope anidado: puede acceder a 'local' y 'global'
    let privada = "soy privada";
    console.log(global);  // ✓ accesible
    console.log(local);   // ✓ accesible (scope padre)
    console.log(privada); // ✓ accesible (propio scope)
  }

  interior();
  // console.log(privada) → ✗ ReferenceError: no accesible aquí
}
Scope (Ámbito): Define dónde son accesibles las variables. JS usa lexical scope: las funciones internas pueden acceder a las variables del contexto externo, pero no al revés.

06 Closure función + entorno léxico
06-closure.js 
// Closure: una función interna que "recuerda" las variables
// de su contexto externo, incluso después de que este haya terminado.

function crearContador() {
  let cuenta = 0; // variable en el scope de crearContador

  return {
    // La función interna tiene acceso a 'cuenta' por closure
    incrementar: () => ++cuenta,
    decrementar: () => --cuenta,
    valor: () => cuenta
  };
}

const contador = crearContador();
contador.incrementar(); // 1
contador.incrementar(); // 2
contador.decrementar(); // 1
console.log(contador.valor()); // 1
Closure: Una función "cierra" sobre las variables de su entorno léxico. Permite crear datos privados y fábricas de funciones. Es uno de los conceptos más poderosos de JavaScript.

07 Recursión función que se llama a sí misma
07-recursion.js 
// Recursión: una función que se llama a sí misma.
// SIEMPRE necesita un caso base para detenerse (evitar stack overflow).

function factorial(n) {
  // Caso base: factorial de 0 o 1 es 1
  if (n <= 1) return 1;

  // Caso recursivo: n * factorial(n-1)
  return n * factorial(n - 1);
}

// factorial(5) → 5 * 4 * 3 * 2 * 1 = 120
console.log(factorial(5)); // 120

// Fibonacci recursivo: cada número = suma de los dos anteriores
const fibonacci = n =>
  n <= 1 ? n : fibonacci(n - 1) + fibonacci(n - 2);
Recursión: Técnica donde una función se invoca a sí misma para dividir un problema en partes más pequeñas. Requiere un caso base para detener la cadena. Muy útil en estructuras jerárquicas y algoritmos de búsqueda.

08 IIFE Immediately Invoked Function Expression
08-iife.js 
// IIFE: función que se define y ejecuta inmediatamente.
// Crea su propio scope, evitando contaminar el scope global.

const resultado = (() => {
  // Todo lo declarado aquí no contamina el scope global
  const privado = "no accesible desde afuera";
  const pi = Math.PI;
  return `π ≈ ${pi.toFixed(4)}`;
})(); // ← los () del final la invocan inmediatamente

console.log(resultado); // "π ≈ 3.1416"
// console.log(privado) → ✗ ReferenceError

// IIFE clásica con parámetros
((nombre, version) => {
  console.log(`App: ${nombre} v${version}`);
})("MiApp", "1.0.0");
IIFE: Se usa para encapsular código de inicialización, crear módulos privados o evitar conflictos de nombres en el scope global. Era muy común antes de los módulos ES6.

09 Funciones de orden superior higher-order functions
09-higher-order.js 
// Función de orden superior: recibe o retorna otras funciones.
// map, filter y reduce son los ejemplos más comunes en arrays.

const productos = [
  { nombre: "Laptop",   precio: 1200 },
  { nombre: "Mouse",    precio: 25   },
  { nombre: "Monitor",  precio: 350  },
  { nombre: "Teclado",  precio: 80   },
];

// filter: filtra elementos según condición
const caros = productos.filter(p => p.precio > 100);

// map: transforma cada elemento
const nombres = caros.map(p => p.nombre);

// reduce: acumula un resultado único
const totalCaros = caros.reduce((sum, p) => sum + p.precio, 0);

console.log(nombres, totalCaros); // ["Laptop","Monitor"], 1550
Funciones de orden superior: Las funciones son "ciudadanos de primera clase" en JS — se pueden pasar como argumentos, retornar y asignar. map transforma, filter selecciona, reduce acumula.
