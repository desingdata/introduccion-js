// Introducción a JavaScript

SINTAXIS en
JavaScript

Es el conjunto de reglas formales que determinan cómo deben escribirse las instrucciones, declaraciones y estructuras del lenguaje para que el motor de ejecución las interprete correctamente. Define la forma válida de variables, funciones, expresiones, operadores, estructuras de control y delimitadores dentro del código.

01

Reglas de Sintaxis

// Conjunto de normas que definen cómo debe escribirse el código JS para ser válido e interpretable por el motor.

01_reglas.js 
// REGLA 1: JavaScript distingue mayúsculas de minúsculas (case-sensitive)
            let nombre = "Ana";        // variable válida
            let Nombre = "Luis";       // diferente variable (N mayúscula)
            
            // REGLA 2: Los espacios en blanco son ignorados (excepto en strings)
            let suma = 5 + 3;           // válido con espacios
            let suma2=5+3;              // también válido sin espacios
            
            // REGLA 3: Las instrucciones terminan con punto y coma (opcional pero recomendado)
            let x = 10;
            let y = 20;
            
            // REGLA 4: Los bloques de código se delimitan con llaves { }
            if (x < y) {
              console.log("x es menor");  // código dentro del bloque
            }
            
            // REGLA 5: El código se ejecuta de arriba hacia abajo, línea por línea
            console.log("Línea 1");      // se ejecuta primero
            console.log("Línea 2");      // se ejecuta después
02

Valores

// Dato concreto que JS puede almacenar, procesar o retornar. Puede ser primitivo (inmutable) o de referencia (objeto/array).

02_valores.js 
// ── VALORES PRIMITIVOS (tipos simples, inmutables) ──────────────────
          
          let texto      = "Hola mundo";    // String  → cadena de texto
          let edad       = 25;              // Number  → número entero o decimal
          let precio     = 9.99;            // Number  → número decimal (float)
          let activo     = true;            // Boolean → solo true o false
          let sinValor   = null;            // Null    → ausencia intencional de valor
          let sinDef     = undefined;       // Undefined → valor no asignado aún
          let simbolo    = Symbol("id");    // Symbol  → identificador único e irrepetible
          let grande     = 9007199254740991n;// BigInt  → entero de precisión arbitraria
          
          // ── VALORES DE REFERENCIA (apuntan a objetos en memoria) ─────────────
          
          let lista   = [1, 2, 3];           // Array  → colección ordenada de valores
          let persona = { nombre: "Ana" };    // Object → colección de pares clave-valor
          
          // Verificar el tipo de un valor con typeof
          console.log(typeof texto);          // → "string"
          console.log(typeof edad);           // → "number"
          console.log(typeof activo);         // → "boolean"

Literales

// Representación directa y fija de un valor en el código fuente; es el valor en sí mismo escrito explícitamente.

03_literales.js 
// LITERALES NUMÉRICOS → valor numérico escrito directamente
          let entero      = 42;
          let decimal     = 3.14;
          let hexadecimal = 0xFF;              // base 16 → 255
          let binario     = 0b1010;           // base 2  → 10
          let octal       = 0o17;             // base 8  → 15
          
          // LITERALES DE CADENA → texto delimitado por comillas
          let simple    = 'comillas simples';
          let doble     = "comillas dobles";
          let plantilla = `template literal: ${entero}`; // permite expresiones con ${}
          
          // LITERALES BOOLEANOS → solo dos valores posibles
          let verdadero = true;
          let falso     = false;
          
          // LITERAL DE ARREGLO → lista de valores entre corchetes
          let colores = ["rojo", "verde", "azul"];
          
          // LITERAL DE OBJETO → pares clave:valor entre llaves
          let auto = {
            marca: "Toyota",                  // propiedad: valor
            año:   2024,
            activo: true
          };
          
          // LITERAL REGEX → patrón de búsqueda entre barras /pattern/
          let patron = /[a-z]+/gi;           // coincide letras, global, insensible
04

Palabras Clave

// Palabras reservadas por el lenguaje con un significado especial predefinido; no pueden usarse como nombres de variables.

04_palabras_clave.js 
// DECLARACIÓN DE VARIABLES
          var   a = 1;    // scope de función, hoisting, evitar su uso moderno
          let   b = 2;    // scope de bloque, reasignable
          const c = 3;    // scope de bloque, NO reasignable (constante)
          
          // CONTROL DE FLUJO
          if (b > a) {     // evalúa una condición
            console.log("mayor");
          } else {         // caso alternativo
            console.log("menor o igual");
          }
          
          // BUCLES (repetición)
          for (let i = 0; i < 3; i++) {    // repite N veces
            console.log(i);
          }
          
          while (a < 5) {               // repite mientras condición sea true
            a++;
          }
          
          // FUNCIONES
          function saludar() {          // define un bloque reutilizable
            return "Hola";              // retorna un valor
          }
          
          // MANEJO DE ERRORES
          try {                          // intenta ejecutar código
            throw new Error("Algo falló");
          } catch (error) {             // captura el error
            console.log(error.message);
          } finally {                  // siempre se ejecuta
            console.log("Fin del bloque");
          }
          
          // OTRAS PALABRAS CLAVE IMPORTANTES
          import { suma } from './utils.js'; // importar módulos
          export default saludar;           // exportar para otros módulos
          typeof, instanceof, in, delete, void; // operadores-palabra
05

Variables

// Contenedor con nombre que almacena un valor en memoria; ese valor puede ser leído o modificado a lo largo del programa.

05_variables.js 
// ── var: scope de función, tiene hoisting (se eleva al inicio) ───────
          var ciudad = "Bogotá";
          ciudad = "Medellín";               // se puede reasignar
          var ciudad = "Cali";               // incluso redeclarar (no recomendado)
          
          // ── let: scope de bloque, no se puede redeclarar ─────────────────────
          let pais = "Colombia";
          pais = "Argentina";               // reasignación: OK
          // let pais = "Chile"  → ERROR: ya fue declarada
          
          // ── const: scope de bloque, valor inmutable (la referencia) ──────────
          const PI = 3.14159;
          // PI = 3.14  → ERROR: no se puede reasignar
          
          const usuario = { nombre: "Carlos" };
          usuario.nombre = "María";          // OK → el objeto es mutable, la referencia no
          
          // ── Scope: donde vive la variable ─────────────────────────────────────
          let global = "soy global";         // accesible en todo el script
          
          function demo() {
            let local = "soy local";         // solo accesible dentro de la función
            console.log(global);             // puede leer la global
          }
          // console.log(local) → ERROR: no existe aquí
          
          // ── Destructuring: extraer valores de objetos/arrays fácilmente ──────
          const [primero, segundo] = [10, 20];   // primero=10, segundo=20
          const { nombre, edad } = usuario;    // extrae propiedades del objeto

Identificadores

// Nombre dado por el programador a una variable, función, clase u otro elemento; sigue reglas estrictas de formato.

06_identificadores.js 
// ── REGLAS DE IDENTIFICADORES ────────────────────────────────────────
          // ✔ Pueden comenzar con: letra, _ (guion bajo), o $ (dólar)
          // ✔ Pueden contener: letras, dígitos, _, $
          // ✘ NO pueden empezar con un número
          // ✘ NO pueden ser palabras reservadas (let, for, if...)
          
          // VÁLIDOS ✔
          let nombre          = "Ana";       // letras minúsculas
          let nombreCompleto  = "Ana Ruiz";  // camelCase (convención JS)
          let _privado        = 42;          // comienza con _
          let $elemento       = null;        // comienza con $ (usado en jQuery)
          let valor2          = 100;         // número al final: OK
          const MAX_SIZE      = 500;         // SCREAMING_SNAKE_CASE para constantes
          
          // INVÁLIDOS ✘ (estos darían SyntaxError)
          // let 2valor   = ...  → empieza con número
          // let mi-variable     → guion medio no permitido
          // let let      = ...  → es palabra reservada
          // let mi var   = ...  → espacio no permitido
          
          // ── CONVENCIONES DE NOMENCLATURA ─────────────────────────────────────
          let      miVariable       = 1;    // camelCase       → variables y funciones
          const    MiClase          = {};   // PascalCase      → clases y constructores
          const    LIMITE_MAX       = 99;   // UPPER_SNAKE     → constantes globales
          function _metodoInterno() {}     // _underscore     → "privado" por convención

Operadores

// Símbolo especial que realiza una operación sobre uno o más valores (operandos) y devuelve un resultado.

07_operadores.js 
// ── ARITMÉTICOS → operaciones matemáticas ────────────────────────────
          console.log(10 + 3);   // 13  → suma
          console.log(10 - 3);   // 7   → resta
          console.log(10 * 3);   // 30  → multiplicación
          console.log(10 / 3);   // 3.3 → división
          console.log(10 % 3);   // 1   → módulo (resto)
          console.log(2 ** 8);   // 256 → exponenciación
          
          // ── ASIGNACIÓN → guardar/actualizar valor ────────────────────────────
          let n = 10;
          n += 5;     // n = n + 5  → 15
          n -= 3;     // n = n - 3  → 12
          n *= 2;     // n = n * 2  → 24
          n++;        // n = n + 1  → 25 (incremento)
          n--;        // n = n - 1  → 24 (decremento)
          
          // ── COMPARACIÓN → retornan true o false ──────────────────────────────
          console.log(5 ==  "5");  // true  → igual (convierte tipo)
          console.log(5 === "5");  // false → estrictamente igual (tipo + valor)
          console.log(5 !=  3);    // true  → diferente
          console.log(5 >   3);    // true  → mayor que
          console.log(5 <=  5);    // true  → menor o igual que
          
          // ── LÓGICOS → combinan condiciones ───────────────────────────────────
          console.log(true && false);  // false → AND (ambos deben ser true)
          console.log(true || false);  // true  → OR (uno basta)
          console.log(!true);          // false → NOT (niega el valor)
          
          // ── TERNARIO → if-else en una línea ──────────────────────────────────
          const resultado = n > 20 ? "grande" : "pequeño"; // condición ? si : no
          
          // ── NULLISH COALESCING → valor por defecto si null/undefined ─────────
          const valor = null ?? "predeterminado";      // "predeterminado"

Expresiones

// Fragmento de código que combina valores, variables y operadores para producir (evaluar) un único valor de retorno.

08_expresiones.js 
// ── EXPRESIONES ARITMÉTICAS → producen un número ─────────────────────
          let area = 5 * 8;                      // evalúa → 40
          let perimetro = 2 * (5 + 8);           // evalúa → 26
          
          // ── EXPRESIONES DE CADENA → producen texto ────────────────────────────
          let saludo = "Hola" + " " + "Mundo";    // → "Hola Mundo" (concatenación)
          let info   = `Área = ${area}`;          // → "Área = 40" (template)
          
          // ── EXPRESIONES LÓGICAS → producen booleano ──────────────────────────
          let esMayor = area > 30 && area < 100; // → true
          
          // ── EXPRESIÓN DE ASIGNACIÓN → devuelve el valor asignado ─────────────
          let x;
          let y = (x = 10);                      // x=10, y=10 (la asignación es una expr.)
          
          // ── EXPRESIÓN DE FUNCIÓN → función como valor ─────────────────────────
          const doble = function(n) {
            return n * 2;                         // función almacenada en variable
          };
          
          // ── ARROW FUNCTION (función flecha) → sintaxis compacta ──────────────
          const triple = n => n * 3;             // equivale a function(n){ return n*3; }
          
          // ── EXPRESIÓN CONDICIONAL (ternaria) ─────────────────────────────────
          const estado = area > 50
            ? "área grande"                       // si es verdadero
            : "área pequeña";                     // si es falso → "área pequeña"
          
          // ── IIFE → expresión de función invocada inmediatamente ──────────────
          const initData = (() => {
            return { version: "1.0" };           // se ejecuta al declararse
          })();

Declaraciones

// Instrucción completa que realiza una acción; a diferencia de la expresión, no produce un valor de retorno directo.

09_declaraciones.js 
// ── DECLARACIÓN DE VARIABLE → crea un espacio en memoria ─────────────
          let   contador = 0;
          const LIMITE   = 10;
          
          // ── DECLARACIÓN DE FUNCIÓN → define un bloque reutilizable ───────────
          function incrementar(valor, paso = 1) {   // paso tiene valor por defecto
            return valor + paso;
          }
          
          // ── DECLARACIÓN DE CLASE → plantilla para crear objetos ──────────────
          class Producto {
            constructor(nombre, precio) {        // inicializador al crear instancia
              this.nombre = nombre;
              this.precio = precio;
            }
            describir() {                        // método de la clase
              return `${this.nombre}: $${this.precio}`;
            }
          }
          const p1 = new Producto("Zapatos", 59.99);
          
          // ── DECLARACIÓN IF/ELSE → bifurcación condicional ─────────────────────
          if (contador < LIMITE) {
            contador = incrementar(contador);
          }
          
          // ── DECLARACIÓN SWITCH → múltiples casos ──────────────────────────────
          switch (contador) {
            case 0:  console.log("cero");  break; // si es 0
            case 1:  console.log("uno");   break; // si es 1
            default: console.log("otro");        // ningún caso anterior
          }
          
          // ── DECLARACIÓN FOR...OF → itera sobre iterables (arrays, strings) ────
          const items = ["a", "b", "c"];
          for (const item of items) {
            console.log(item);                   // "a", "b", "c"
          }

Comentarios

// Texto en el código que el motor JS ignora completamente; sirven para documentar, explicar y organizar el código fuente.

10_comentarios.js 
// ── COMENTARIO DE UNA LÍNEA → desde // hasta el final de la línea ────
          let base = 100;  // valor base del cálculo
          
          /* ── COMENTARIO MULTILÍNEA → entre /* y */ para bloques más largos ─ */
          /*
            Función que calcula el descuento final.
            Parámetros:
              precio   → precio original del producto
              pct      → porcentaje de descuento (0-100)
            Retorna:
              precio con descuento aplicado
          */
          function descuento(precio, pct) {
            return precio * (1 - pct / 100);
          }
          
          // ── JSDOC → comentario estándar para documentación automatizada ───────
          /**
           * Calcula el IVA de un precio dado.
           * @param {number} precio  - Precio base sin impuesto
           * @param {number} tasa    - Porcentaje IVA (ej. 19 para Colombia)
           * @returns {number}       - Monto del IVA
           */
          function calcularIVA(precio, tasa) {
            return precio * (tasa / 100);         // resultado: ej. 1000 * 0.19 = 190
          }
          
          // ── USO PRÁCTICO: comentar código temporalmente ───────────────────────
          let total = descuento(1000, 10);      // → 900
          // let total = 1000;                 // versión anterior (desactivada)
          
          // ── BUENAS PRÁCTICAS ─────────────────────────────────────────────────
          // ✔ Explicar el PORQUÉ, no el QUÉ (el código ya dice qué hace)
          // ✔ Mantener comentarios actualizados al modificar el código
          // ✔ Usar JSDoc en funciones públicas o de API
          // ✘ Evitar comentarios obvios: let x = 5; // asigna 5 a x
