Operadores Aritméticos
Realizan cálculos matemáticos entre valores numéricos: suma, resta, multiplicación, división, módulo, exponenciación e incremento/decremento.
// ─────────────────────────────────────────────
// OPERADORES ARITMÉTICOS
// Permiten realizar operaciones matemáticas
// básicas entre valores de tipo número.
// ─────────────────────────────────────────────

let a = 10;
let b = 3;

// + Suma: une o suma dos valores
console.log(a + b);          // → 13

// - Resta: calcula la diferencia entre dos valores
console.log(a - b);          // → 7

// * Multiplicación: producto de dos valores
console.log(a * b);          // → 30

// / División: cociente entre dos valores
console.log(a / b);          // → 3.333...

// % Módulo (residuo): resto de la división entera
console.log(a % b);          // → 1

// ** Exponenciación: eleva a la potencia indicada
console.log(a ** b);         // → 1000  (10³)

// ++ Incremento: aumenta el valor en 1
a++;
console.log(a);              // → 11

// -- Decremento: disminuye el valor en 1
b--;
console.log(b);              // → 2

Operadores de Asignación
Asignan un valor a una variable. Combinados con operadores aritméticos, actualizan el valor actual de forma abreviada.
// ─────────────────────────────────────────────
// OPERADORES DE ASIGNACIÓN
// Guardan un valor en una variable.
// Las formas combinadas son atajos de escritura.
// ─────────────────────────────────────────────

let x = 20;   // = Asignación simple: asigna 20 a x

x += 5;       // += Suma y asigna: x = x + 5  → 25
x -= 3;       // -= Resta y asigna: x = x - 3  → 22
x *= 2;       // *= Multiplica y asigna: x = x * 2  → 44
x /= 4;       // /= Divide y asigna: x = x / 4  → 11
x %= 4;       // %= Módulo y asigna: x = x % 4  → 3
x **= 3;      // **= Potencia y asigna: x = x ** 3  → 27

console.log("Resultado final:", x);  // → 27

// Asignación con destructuring (ES6+)
// Extrae valores de arrays u objetos directamente
const [primero, segundo] = [100, 200];
console.log(primero, segundo);       // → 100  200

Operadores de Comparación
Comparan dos valores y devuelven un booleano (true / false). Es clave distinguir == (igual en valor) de === (igual en valor Y tipo).
// ─────────────────────────────────────────────
// OPERADORES DE COMPARACIÓN
// Evalúan la relación entre dos valores.
// Siempre retornan true o false.
// ─────────────────────────────────────────────

let p = 5;
let q = "5";
let r = 8;

// == Igualdad abstracta: compara valor, ignora tipo
console.log(p == q);          // → true  (5 == "5")

// === Igualdad estricta: compara valor Y tipo de dato
console.log(p === q);         // → false (number ≠ string)

// != Desigualdad abstracta
console.log(p != r);          // → true

// !== Desigualdad estricta (valor o tipo diferente)
console.log(p !== q);         // → true  (tipos distintos)

// > Mayor que
console.log(r > p);           // → true

// < Menor que
console.log(p < r);           // → true

// >= Mayor o igual que
console.log(p >= 5);          // → true

// <= Menor o igual que
console.log(r <= 8);          // → true

Operadores Lógicos
Combinan o invierten expresiones booleanas. Son la base de toda lógica condicional en JavaScript.
// ─────────────────────────────────────────────
// OPERADORES LÓGICOS
// Combinan condiciones y retornan boolean.
// Fundamentales para estructuras if/else y bucles.
// ─────────────────────────────────────────────

let esMayor  = true;
let tieneID  = false;
let esSocio  = true;

// && AND lógico: ambas condiciones deben ser true
console.log(esMayor && tieneID);   // → false
console.log(esMayor && esSocio);    // → true

// || OR lógico: al menos una condición debe ser true
console.log(esMayor || tieneID);   // → true
console.log(false   || false);     // → false

// ! NOT lógico: invierte el valor booleano
console.log(!esMayor);             // → false
console.log(!tieneID);             // → true

// ?? Nullish coalescing: usa valor derecho si el
//    izquierdo es null o undefined
let nombre = null;
console.log(nombre ?? "Invitado");  // → "Invitado"

// Operador ternario: condición ? valorSi : valorNo
let acceso = esMayor && esSocio
             ? "Permitido"
             : "Denegado";
console.log(acceso);                // → "Permitido"

Aritmética de JavaScript
El objeto Math proporciona constantes y métodos matemáticos avanzados. Además, JS tiene comportamientos especiales como NaN, Infinity y la imprecisión de punto flotante.
// ─────────────────────────────────────────────
// ARITMÉTICA DE JAVASCRIPT
// JS usa IEEE 754 (punto flotante de 64 bits).
// El objeto Math ofrece funciones matemáticas.
// ─────────────────────────────────────────────

// Casos especiales del motor numérico
console.log(0.1 + 0.2);           // → 0.30000000000000004 (float!)
console.log(1 / 0);               // → Infinity
console.log("hola" * 2);         // → NaN (Not a Number)
console.log(isNaN("hola" * 2));   // → true  (verifica NaN)

// Constantes matemáticas útiles
console.log(Math.PI);             // → 3.14159...  (π)
console.log(Math.E);              // → 2.71828...  (Euler)

// Métodos del objeto Math
console.log(Math.round(4.7));     // → 5   (redondea al entero más cercano)
console.log(Math.floor(4.9));     // → 4   (redondea hacia abajo)
console.log(Math.ceil(4.1));      // → 5   (redondea hacia arriba)
console.log(Math.abs(-15));       // → 15  (valor absoluto)
console.log(Math.sqrt(49));       // → 7   (raíz cuadrada)
console.log(Math.pow(2, 10));    // → 1024 (potencia)
console.log(Math.max(3,9,1));    // → 9   (máximo de un conjunto)
console.log(Math.min(3,9,1));    // → 1   (mínimo de un conjunto)

// Número aleatorio entre 1 y 100
let aleatorio = Math.floor(Math.random() * 100) + 1;
console.log("Aleatorio:", aleatorio);  // → número entre 1-100
