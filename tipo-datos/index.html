Tipos de Datos

// Fundamentos y estructuras base del lenguaje

¿Qué son los tipos de datos?

Clasificación que define qué tipo de valor puede almacenar una variable: número, texto, booleano, objeto, etc.

// Cada valor tiene un tipo de dato asociado
let nombre  = "Ana";         // string
let edad    = 25;           // number
let activo  = true;         // boolean
let nada    = null;         // null
let sin_def = undefined;   // undefined
let obj     = { a: 1 };     // object

// typeof devuelve el tipo como string
console.log(typeof nombre);  // "string"
console.log(typeof edad);    // "number"
Estructura de Datos

Forma en que organizamos y almacenamos datos para acceder y modificarlos eficientemente.

// PRIMITIVOS: valor único, inmutables
let x = 42;
let y = x;   // copia por valor
y = 100;     // x sigue siendo 42

// REFERENCIA: arrays y objetos
let arr1 = [1, 2, 3];
let arr2 = arr1;  // copia por referencia
arr2.push(4);     // arr1 también cambia!
console.log(arr1); // [1, 2, 3, 4]
Falsy y Truthy

En JS, valores que se evalúan como false (falsy) o true (truthy) en contextos booleanos.

// FALSY: valores que son "falsos"
if (false) {}   // false
if (0) {}       // 0, -0, 0n
if ("") {}      // string vacío
if (null) {}    // null
if (undefined) {} // undefined
if (NaN) {}     // NaN

// TRUTHY: todo lo demás
if ("0") {}    // string no vacío → true
if ([]) {}     // array vacío → true
if ({}) {}     // objeto vacío → true
Cadena de Prototipos

Mecanismo de herencia en JS donde los objetos heredan propiedades y métodos de su prototipo.

// Cada objeto tiene un [[Prototype]]
const animal = {
  hablar() { return "..."; }
};
const perro = Object.create(animal);
perro.nombre = "Rex";

// perro hereda hablar() de animal
perro.hablar(); // "..."

// __proto__ apunta al prototipo
perro.__proto__ === animal; // true
Symbols en JavaScript

Tipo de dato primitivo único e inmutable, ideal como clave de propiedades para evitar colisiones.

// Symbol crea un identificador único
const id1 = Symbol("id");
const id2 = Symbol("id");
id1 === id2; // false — siempre únicos

// Uso como clave de objeto
const KEY = Symbol("clave");
const obj = { [KEY]: "valor secreto" };
obj[KEY]; // "valor secreto"

// No aparece en for...in ni Object.keys
Object.keys(obj); // [] vacío
Using: Limpieza de Recursos

Propuesta (Stage 3) que permite liberar recursos automáticamente al salir del bloque con using.

// using llama a [Symbol.dispose] al terminar
class Conexion {
  [Symbol.dispose]() {
    console.log("Conexión cerrada");
  }
}

// El recurso se libera automáticamente
{
  using conn = new Conexion();
  // ... usar conn ...
} // → "Conexión cerrada" aquí

Number

// Números, matemáticas y tipos especiales

number.js 
// ─────────────────────────────────────────────────────────────────
// ¿QUÉ ES UN NUMBER?
// Tipo primitivo para representar valores numéricos (enteros y
// decimales). JS usa el estándar IEEE 754 de 64 bits (doble precisión).
// ─────────────────────────────────────────────────────────────────

let entero   = 42;
let decimal  = 3.14;
let negativo = -7;
let infinito = Infinity;   // resultado de 1/0
let hex      = 0xFF;      // hexadecimal → 255
let octal    = 0o17;     // octal → 15
let binario  = 0b1010;   // binario → 10


// ─────────────────────────────────────────────────────────────────
// ¿QUÉ ES NaN (Not a Number)?
// Valor especial que indica resultado de operación matemática inválida.
// Es el único valor en JS que no es igual a sí mismo.
// ─────────────────────────────────────────────────────────────────

let invalido = 0 / 0;              // NaN
let texto_mal = parseInt("abc");   // NaN
let raiz_neg = Math.sqrt(-1);     // NaN

console.log(NaN === NaN);        // false — peculiaridad de JS
console.log(isNaN(invalido));    // true  — usar isNaN()
console.log(Number.isNaN(invalido)); // true — versión más segura


// ─────────────────────────────────────────────────────────────────
// CONVERSIONES NUMÉRICAS
// Transformar otros tipos de dato a Number usando distintos métodos.
// ─────────────────────────────────────────────────────────────────

Number("42");        // 42   — conversión explícita
Number("");          // 0    — string vacío → 0
Number(true);        // 1    — booleano
Number(false);       // 0
Number(null);        // 0
Number(undefined);  // NaN
Number("3.7abc");   // NaN  — no puede convertir

parseInt("3.7abc");  // 3    — lee hasta que no puede
parseFloat("3.7abc"); // 3.7  — idem con decimales
+"42";               // 42   — operador unario +


// ─────────────────────────────────────────────────────────────────
// EL OBJETO MATH
// Objeto global con constantes y funciones matemáticas estáticas.
// ─────────────────────────────────────────────────────────────────

Math.PI;              // 3.14159... → constante π
Math.round(4.6);     // 5   → redondeo al más cercano
Math.floor(4.9);     // 4   → redondeo hacia abajo
Math.ceil(4.1);      // 5   → redondeo hacia arriba
Math.abs(-7);        // 7   → valor absoluto
Math.sqrt(16);       // 4   → raíz cuadrada
Math.pow(2, 8);       // 256 → potencia
Math.max(1, 5, 3);   // 5   → máximo
Math.min(1, 5, 3);   // 1   → mínimo
Math.random();       // 0.X → número aleatorio [0, 1)
Math.trunc(4.9);     // 4   → parte entera (sin redondear)

// Número aleatorio entre min y max
const randInt = (min, max) =>
  Math.floor(Math.random() * (max - min + 1)) + min;
randInt(1, 10); // número entre 1 y 10


// ─────────────────────────────────────────────────────────────────
// EL TIPO BigInt
// Permite representar enteros más grandes que Number.MAX_SAFE_INTEGER
// (2^53 - 1). Se define añadiendo 'n' al final del número.
// ─────────────────────────────────────────────────────────────────

const grande = 9007199254740991n;  // BigInt con 'n'
const otro   = BigInt("12345678901234567890");

grande + 10n;          // operaciones con BigInt
typeof grande;          // "bigint"
// ⚠️ No mezclar BigInt con Number directamente:
// grande + 10; → TypeError
Number(grande) + 10;   // convertir antes si es necesario

Strings

// Cadenas de texto, búsqueda y manipulación

strings.js 
// ─────────────────────────────────────────────────────────────────
// ¿QUÉ ES UN STRING?
// Secuencia de caracteres Unicode (texto). Es inmutable: no se
// puede cambiar un carácter directamente, sino crear uno nuevo.
// ─────────────────────────────────────────────────────────────────

let simple   = 'Hola mundo';           // comillas simples
let doble    = "Hola mundo";           // comillas dobles
let template = `Hola ${"mundo"}`;     // template literal
let multi    = `Línea 1
Línea 2`;                             // multilínea

// Propiedades y acceso básico
"Hola".length;  // 4 — número de caracteres
"Hola"[0];      // "H" — acceso por índice
"Hola".at(-1);  // "a" — índice negativo (nuevo)


// ─────────────────────────────────────────────────────────────────
// POSICIONES Y SUBSTRINGS
// Obtener partes de una cadena según posición o índices.
// ─────────────────────────────────────────────────────────────────

const frase = "JavaScript es genial";

// indexOf / lastIndexOf → posición de una subcadena
frase.indexOf("es");       // 11 — primera ocurrencia
frase.lastIndexOf("a");   // 17 — última ocurrencia
frase.indexOf("python");  // -1  — no encontrado

// slice(inicio, fin) → extrae subcadena
frase.slice(0, 10);    // "JavaScript"
frase.slice(-6);       // "genial" — desde el final

// substring(inicio, fin) → similar a slice, sin negativos
frase.substring(0, 10); // "JavaScript"


// ─────────────────────────────────────────────────────────────────
// BUSCAR Y REEMPLAZAR
// Encontrar patrones y sustituirlos con nuevos valores.
// ─────────────────────────────────────────────────────────────────

const texto = "El gato y el gato negro";

// includes → ¿contiene la subcadena?
texto.includes("gato");    // true

// startsWith / endsWith
texto.startsWith("El");   // true
texto.endsWith("negro");  // true

// replace → reemplaza primera ocurrencia
texto.replace("gato", "perro");
// "El perro y el gato negro"

// replaceAll → reemplaza todas las ocurrencias
texto.replaceAll("gato", "perro");
// "El perro y el perro negro"

// Con expresión regular (g = global)
texto.replace(/gato/g, "perro");
// "El perro y el perro negro"


// ─────────────────────────────────────────────────────────────────
// MODIFICAR CADENAS DE TEXTO
// Transformaciones como cambiar case, eliminar espacios, unir, etc.
// Los strings son inmutables: cada método devuelve uno nuevo.
// ─────────────────────────────────────────────────────────────────

let s = "  Hola Mundo  ";

s.toUpperCase();  // "  HOLA MUNDO  " — todo mayúsculas
s.toLowerCase();  // "  hola mundo  " — todo minúsculas
s.trim();         // "Hola Mundo"     — elimina espacios
s.trimStart();    // "Hola Mundo  "   — solo al inicio
s.trimEnd();      // "  Hola Mundo"   — solo al final

// padStart / padEnd → rellenar hasta longitud
"5".padStart(3, "0");  // "005" — útil para formatos
"hi".padEnd(5, ".");   // "hi..."

// repeat → repetir la cadena N veces
"ha".repeat(3);         // "hahaha"

// split → dividir en array por separador
"a,b,c".split(",");     // ["a", "b", "c"]

// join (en array) → unir array en string
["a", "b", "c"].join("-"); // "a-b-c"


// ─────────────────────────────────────────────────────────────────
// CONTENIDO DE TEXTO UNICODE
// JavaScript usa UTF-16. Algunos caracteres (emoji, símbolos raros)
// usan 2 unidades de código (surrogate pairs).
// ─────────────────────────────────────────────────────────────────

// charCodeAt / codePointAt → código del carácter
"A".charCodeAt(0);     // 65
"😊".codePointAt(0);  // 128522 (requiere codePointAt)

// fromCharCode / fromCodePoint → carácter desde código
String.fromCharCode(65);       // "A"
String.fromCodePoint(128522); // "😊"

// normalize → formas de normalización Unicode (NFC, NFD, etc.)
"\u00e9".normalize("NFC");  // "é" — forma compuesta

// ⚠️ length puede sorprender con emojis
"😊".length; // 2 — ocupa 2 unidades UTF-16
[..."😊"].length; // 1 — spread itera por codepoints

Operadores

// Comparación, lógica y operadores especiales

Operadores Básicos

Operadores aritméticos y de asignación para realizar cálculos y asignar valores.

// Aritméticos
5 + 3;    // 8  suma
5 - 3;    // 2  resta
5 * 3;    // 15 multiplicación
10 / 4;   // 2.5 división
10 % 3;   // 1  módulo (resto)
2 ** 10;  // 1024 potencia

// Asignación compuesta
let x = 5;
x += 3;   // x = x + 3 → 8
x++;      // x = x + 1 → 9 (post-increment)
x--;      // x = x - 1 → 8 (post-decrement)
Operadores de Comparación

Comparan dos valores y devuelven un booleano. La diferencia entre == y === es crucial.

// Igualdad abstracta (==) — hace coerción
1 == "1";    // true  ← peligroso
null == undefined; // true

// Igualdad estricta (===) — sin coerción
1 === "1";   // false ← recomendado
1 === 1;     // true

// Relacionales
5 > 3;       // true
5 >= 5;      // true
3 < 5;       // true
3 !== 4;     // true — distinto estricto

// ✅ Usar siempre === y !==
Operadores Lógicos

Operan sobre valores booleanos (o cualquier valor usando truthy/falsy). Devuelven uno de los operandos.

// AND (&&) — si izq es falsy, lo devuelve
true && true;   // true
0 && "hola";    // 0 (cortocircuito)
"a" && "b";    // "b"

// OR (||) — si izq es truthy, lo devuelve
false || true; // true
0 || "alt";    // "alt" — valor por defecto
"a" || "b";   // "a"

// NOT (!) — invierte el booleano
!true;         // false
!!"texto";     // true — coerción a booleano
Nullish Coalescing MID

Operador ?? que devuelve el operando derecho solo si el izquierdo es null o undefined (no otros falsy).

// ?? vs || — diferencia clave
const a = 0 || "defecto";   // "defecto" (0 es falsy)
const b = 0 ?? "defecto";   // 0 (no es null/undefined)

const c = null ?? "valor";  // "valor"
const d = undefined ?? 42; // 42
const e = "" ?? "nada";     // "" — string vacío se respeta

// ??= asignación nullish
let config = null;
config ??= { modo: "oscuro" };
// config = { modo: "oscuro" }
Operadores BitWise MID

Operan a nivel de bits sobre enteros de 32 bits. Útiles en criptografía, gráficos o flags.

// Convierte a entero 32 bits y opera bit a bit
5 & 3;   // 1   — AND:  0101 & 0011 = 0001
5 | 3;   // 7   — OR:  0101 | 0011 = 0111
5 ^ 3;   // 6   — XOR: 0101 ^ 0011 = 0110
~5;      // -6  — NOT: ~0101 = ...11111010

// Desplazamiento de bits
5 << 1;  // 10  — shift izquierda (x2)
20 >> 2; // 5   — shift derecha (÷4)

// Truco: ~~ como Math.trunc rápido
~~4.9;   // 4 — parte entera
Optional Chaining (?.)

Accede a propiedades anidadas sin lanzar error si algún valor intermedio es null o undefined.

const user = {
  nombre: "Ana",
  direccion: {
    ciudad: "Madrid"
  }
};

// Sin ?. → TypeError si falta propiedad
// user.telefono.numero → 💥 Error

// Con ?. → devuelve undefined si no existe
user?.telefono?.numero;  // undefined
user?.direccion?.ciudad; // "Madrid"

// Con métodos y arrays
user?.getNombre?.();   // llama si existe
user?.tags?.[0];       // primer elemento si existe

// Combinar con ??
user?.edad ?? 18; // 18 si edad es nullish

Estructuras Avanzadas

// Map, Set, WeakMap y colecciones especiales

estructuras-avanzadas.js 
// ─────────────────────────────────────────────────────────────────
// Map — Diccionario clave/valor donde las claves pueden ser
// de cualquier tipo (incluso objetos o funciones).
// ─────────────────────────────────────────────────────────────────

const mapa = new Map();

mapa.set("nombre", "Ana");  // añadir clave string
mapa.set(42, "respuesta");    // clave numérica
mapa.set(true, "activo");    // clave booleana

mapa.get("nombre");   // "Ana"
mapa.has(42);         // true
mapa.delete(true);    // elimina la clave true
mapa.size;            // 2

// Iterar Map
for (const [clave, valor] of mapa) {
  console.log(clave, valor);
}


// ─────────────────────────────────────────────────────────────────
// Set — Colección de valores únicos. Elimina duplicados automáticamente.
// Útil para filtrar, hacer uniones e intersecciones.
// ─────────────────────────────────────────────────────────────────

const set = new Set([1, 2, 2, 3, 3, 3]);
console.log(set); // Set {1, 2, 3} — sin duplicados

set.add(4);       // añade 4
set.has(2);       // true
set.delete(1);    // elimina 1
set.size;         // 3

// Truco: eliminar duplicados de un array
const arr = [1, 1, 2, 3, 2];
const unicos = [...new Set(arr)]; // [1, 2, 3]


// ─────────────────────────────────────────────────────────────────
// WeakMap y WeakSet — Versiones "débiles" de Map/Set.
// Sus referencias NO impiden la recolección de basura (GC).
// Solo aceptan objetos como claves/valores (no primitivos).
// ─────────────────────────────────────────────────────────────────

let obj = { id: 1 };
const wm = new WeakMap();
wm.set(obj, "datos privados");

obj = null; // el objeto puede ser recogido por GC
// wm ya no mantiene la referencia → memoria liberada

// Uso típico: datos privados de clases
const _privados = new WeakMap();
class Persona {
  constructor(nombre, edad) {
    _privados.set(this, { nombre, edad });
  }
  saludo() {
    const { nombre } = _privados.get(this);
    return `Hola, soy ${nombre}`;
  }
}
